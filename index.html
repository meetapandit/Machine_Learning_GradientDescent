"""
Function
--------
bivariate_ols
    Gradient Decent to minimize OLS. Used to find co-efficients of bivariate OLS Linear regression

Parameters
----------
xvalues, yvalues : narray
    xvalues: independent variable
    yvalues: dependent variable
    
R: float
    Learning rate
    
MaxIterations: Int
    maximum number of iterations
    

Returns
-------
alpha: float
    intercept
    
beta: float
    co-efficient
"""
import time

def bivariate_ols(xvalue_matrix, yvalues, R=0.01, MaxIterations=10000):
    start_time = time.time()
    x = xvalue_matrix
    #Calculate the length of target vector
    m = len(yvalues)
    #get the length of the output variable
    n = len(x.T)
    
    #initialize coefficients vector with all 10 values
    coeff = np.array([10]*n)
    #Convert to matrix
    coeff = coeff.reshape(-1,1)
    
    #transpose the matrix of predictors
    xTrans = x.transpose()
    
    #Convert the vector of traget to a matrix
    y = yvalues.values.reshape(-1,1)
    #dot product of predictors and coefficient matrix gives the predicted values
    yhat = np.dot(x,coeff)
    #Calculate error
    e = yhat - y
    
    #sum of squared error(cost) from the error matrix obtained above
    J = np.sum(e**2)
    
    #set converged to False by default
    converged=False
    iteration=0
    while not converged:
        #Calculate gradient
        gradient = np.dot(xTrans,e)/m
        
        #Calculate new coefficient values with learning rate and gradient
        coeff = coeff - R*gradient
        #Update cost value with new coefficient values
        yhat = np.dot(x,coeff)
        e=yhat-y
        cost = np.sum(e**2)
        
        #Check if we have reached the convergence condition
        if(abs(cost-J)<=0.0001):
            print("Converged after iteration: %s"%iteration)
            converged = True
        else:
            J=cost
        #condition for max number of iterations        
        if iteration==MaxIterations:
            print("Max Iterations Reached")
            converged = True
        iteration+=1
    print ("Time taken: " + str(round(time.time() - start_time,2)) + " seconds")
        
    return coeff[0], coeff[1]

def compute_rmse(prediction,yvalues):
    if len(np.array(prediction))==len(np.array(yvalues)):
        rmse = np.sqrt(np.mean((np.array(prediction)-np.array(yvalues))**2))
    else:
        print("array length not equal")
        return -1
    return rmse

import time
def ridge_multivariate_ols(xvalue_matrix, yvalues, R, L, MaxIterations):
    start_time = time.time()
    x = xvalue_matrix
    #Calculate the length of target vector
    m = len(yvalues)
    #get the length of the output variable
    n = len(x.T)
    
    #initialize coefficients vector with all 10 values
    coeff = np.array([10]*n)
    #Convert to matrix
    coeff = coeff.reshape(-1,1)
    
    #transpose the matrix of predictors
    xTrans = x.transpose()
    
    #Convert the vector of traget to a matrix
    y = yvalues.values.reshape(-1,1)
    #dot product of predictors and coefficient matrix gives the predicted values
    yhat = np.dot(x,coeff)
    #Calculate error
    e = yhat - y
    
    #sum of squared error(cost) from the error matrix obtained above
    J = np.sum(e**2)
    
    #set converged to False by default
    converged=False
    iteration=0
    while not converged:
        #Calculate gradient
        gradient = np.dot(xTrans,e)/m
        
        #Calculate new coefficient values with learning rate and gradient
        coeff = coeff - R*(gradient+ (L/m)*coeff)
        #Update cost value with new coefficient values
        yhat = np.dot(x,coeff)
        e=yhat-y
        cost = np.sum(e**2)
        
        #Check if we have reached the convergence condition
        if(abs(cost-J)<=0.0001):
            print("Converged after iteration: %s"%iteration)
            converged = True
        else:
            J=cost
        #condition for max number of iterations        
        if iteration==MaxIterations:
            print("Max Iterations Reached")
            converged = True
        iteration+=1
    print ("Time taken: " + str(round(time.time() - start_time,2)) + " seconds")
        
    return coeff[0], coeff[1:]
